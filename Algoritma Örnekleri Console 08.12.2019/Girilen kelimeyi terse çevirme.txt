using System;

namespace Algoritma
{
    class Program
    {
        public static void Main(string[] args)
        {
            String metin;
            Console.Write("Tersten Yazýlacak Metni Giriniz = ");
            metin = Console.ReadLine();
            for (int i = 0; i <= metin.Length - 1; i++)
            {
                Console.Write(metin[metin.Length - i - 1]);
            }
            Console.ReadLine();
        }
    }
}
